	function GetSQLValueString($theValue, $theType, $theDefinedValue = "", $theNotDefinedValue = "")

		case "int":
		$theValue = ($theValue != "") ? intval($theValue) : "NULL";

		$theValue = ($theValue != "") ? "'" . $theValue . "'" : "NULL";
		break;
		case "defined":
		$theValue = ($theValue != "") ? $theDefinedValue : $theNotDefinedValue;
		break;
	}
	return $theValue;
	}
